/**
 * Light Base Class
 */
 
import { LightData, LightType, RGB } from '../types';
import { Vector3 } from '../../geometry/Vector3';
 
export abstract class Light {
  public id: string;
  public name: string;
  public type: LightType;
  public color: RGB;
  public intensity: number;
  public enabled: boolean;
  
  constructor(id: string, name: string, type: LightType) {
    this.id = id;
    this.name = name;
    this.type = type;
    this.color = { r: 255, g: 255, b: 255 };
    this.intensity = 1.0;
    this.enabled = true;
  }
  
  public setColor(r: number, g: number, b: number): this {
    this.color.r = Math.max(0, Math.min(255, Math.round(r)));
    this.color.g = Math.max(0, Math.min(255, Math.round(g)));
    this.color.b = Math.max(0, Math.min(255, Math.round(b)));
    return this;
  }
  
  public setIntensity(value: number): this {
    this.intensity = Math.max(0, Math.min(10, value));
    return this;
  }
  
  public setEnabled(enabled: boolean): this {
    this.enabled = enabled;
    return this;
  }
  
  public abstract calculateIllumination(point: Vector3): number;
  
  public abstract clone(): Light;
  
  protected cloneBase(light: Light): void {
    light.color = { ...this.color };
    light.intensity = this.intensity;
    light.enabled = this.enabled;
  }
}
 
export class Light_IO {
  public static serialize(light: Light): LightData {
    return {
      id: light.id,
      name: light.name,
      type: light.type,
      color: { ...light.color },
      intensity: light.intensity,
      enabled: light.enabled
    };
  }
  
  public static serializeBase(light: Light): LightData {
    return Light_IO.serialize(light);
  }
}
